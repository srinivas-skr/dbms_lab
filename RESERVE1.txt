
CREATE TABLE Sailors (
    sid INTEGER,
    sname VARCHAR(255),
    rating INTEGER,
    age REAL,
    PRIMARY KEY (sid)
);

CREATE TABLE Boats (
    bid INTEGER,
    bname VARCHAR(255),  
    color VARCHAR(255),  
    PRIMARY KEY (bid)
);


CREATE TABLE Reserves (
    sid INTEGER,
    bid INTEGER,
    day DATE,
    FOREIGN KEY (sid) REFERENCES Sailors(sid),
    FOREIGN KEY (bid) REFERENCES Boats(bid)
);
